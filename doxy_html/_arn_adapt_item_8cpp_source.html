722 void ArnAdaptItem::arnEvent( QEvent* ev, bool isAlienThread)
723 {
724  Q_UNUSED(isAlienThread)
725  Q_D(ArnAdaptItem);
726 
727  //int evIdx = ev->type() - ArnEvent::baseType();
728  int evIdx = ArnEvent::toIdx( ev->type());
729 
730  if (d->_arnEventCB) {
731  (*(d->_arnEventCB))( ev, evIdx);
732  }
733 
734  switch (evIdx) {
735  case ArnEvent::Idx::ValueChange:
736  {
737  if (!d->_changedCB) return;
738  ArnEvValueChange* e = static_cast<ArnEvValueChange*>( ev);
739  ArnAdaptItem* target = static_cast<ArnAdaptItem*>( e->target());
740  if (!target) return; // No target, deleted/closed ...
741 
742  QByteArray val = e->valueData() ? *e->valueData() : target->toByteArray();
743  (*(d->_changedCB))( *target, val);
744  // qDebug() << "ArnAdaptEvValueChange: path=" << target->path()
745  // << " value=" << val;
746  return;
747  }
748  case ArnEvent::Idx::ModeChange:
749  {
750  ArnEvModeChange* e = static_cast<ArnEvModeChange*>( ev);
751  ArnAdaptItem* target = static_cast<ArnAdaptItem*>( e->target());
752  if (!target) return; // No target, deleted/closed ...
753 
754  QMutexLocker mutexLocker( &target->mutex()); // Force atomic operation on target
755 
756  // qDebug() << "ArnAdaptEvModeChange: path=" << e->path() << " mode=" << e->mode()
757  // << " inItemPath=" << target->path();
758  if (!target->isFolder()) {
759  if (e->mode().is( Arn::ObjectMode::Pipe)) { // Pipe-mode never IgnoreSameValue
760  target->setIgnoreSameValue(false);
761  }
762  }
763  return;
764  }
765  case ArnEvent::Idx::Retired:
766  {
767  ArnEvRetired* e = static_cast<ArnEvRetired*>( ev);
768  ArnAdaptItem* target = static_cast<ArnAdaptItem*>( e->target());
769  if (!target) return; // No target, deleted/closed ...
770 
771  QMutexLocker mutexLocker( &target->mutex()); // Force atomic operation on target
772 
773  if (!e->isBelow()) {
774  if (Arn::debugLinkDestroy) qDebug() << "AdaptItem arnLinkDestroyed: path=" << target->path();
775  if (d->_linkDestroyedCB) {
776  (*(d->_linkDestroyedCB))( *target);
777  }
778  target->close();
779  e->setTarget(0); // target is not available any more
780  }
781  return;
782  }
783  default:;
784  }
785 }
