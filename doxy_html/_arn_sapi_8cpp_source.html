68 {
69  Q_D(ArnSapi);
70 
71  if (mode.is( mode.Provider)) {
72  d->_receivePrefix = providerPrefix ? providerPrefix : "pv_";
73  d->_sendPrefix = requesterPrefix ? requesterPrefix : "rq_";
